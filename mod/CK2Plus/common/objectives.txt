# Gain one of a vassal's landed titles
plot_gain_vassal_title = {
	type = vassal_titles
	plot = yes
	vassal_rank_plot = yes
	
	# Plotter scope
	potential = {
		is_playable = yes
		prisoner = no
		age = 16
		primary_title = { higher_tier_than = COUNT }
		NOT = { trait = incapable }
		NOT = { trait = imbecile }
		
		OR = {
			ai = no
			is_primary_war_defender = no # Do not initiate this plot while we´re in a defensive war
		}
	}
	
	# Target scope
	allow = {
		# No barony titles
		higher_tier_than = BARON
		is_primary_type_title = no
		controls_religion = no
		mercenary = no
		OR = {
		
			# Count title outside a Duke's duchy
			AND = {
				tier = COUNT
				FROM = { demesne_efficiency = 1.01 }
				holder_scope = {
					tier = DUKE
					ROOT = {
						NOT = { de_jure_liege = PREV }
					}
				}
			}
			# Duke with no count titles
			AND = {
				tier = DUKE
				holder_scope = {
					NOT = {
						any_demesne_title = {
							tier = COUNT
							location = {
								duchy = {
									title = ROOT
								}
							}
						}
					}
					NOT = {
						any_vassal = {
							any_demesne_title = {

									duchy = {
										title = ROOT
									}
								}
							}
						}
					}
				}
			}
			# Count titles in ruler's primary duchy
			AND = {
				tier = COUNT
				FROM = { demesne_efficiency = 1.01 }
				location = {
					duchy = {
						holder_scope = {
							character = FROM
						}
						FROM = {
							capital_scope = {
								duchy = {
									title = PREVPREVPREV
								}
							}
						}
					}
				}
			}
			# Count's secondary county
			AND = {
				tier = COUNT
				FROM = { demesne_efficiency = 1.01 }
				location = {
					is_capital = no
				}
				holder_scope = {
					lower_tier_than = DUKE
					num_of_count_titles = 2
				}
			}
		}
	}

		factor = 5
		
		modifier = {
			factor = 0.0
			FROM = {
				ai = yes
				war = yes
			}
		}

			factor = 0.01
			FROM = { trait = content }
		}

			factor = 0.2
			FROM = { trait = slow }
		}

			factor = 0.2
			FROM = { trait = kind }
		}

			factor = 0.0
			FROM = {
				ai = yes
				family = ROOT
				opinion = { who = ROOT value = 25 }
			}
		}
		modifier = {
			factor = 0
			FROM = {
				ai = yes
				opinion = { who = ROOT value = 50 }
			}
		}
		modifier = {
			factor = 0.1
			FROM = {
				ai = yes
				opinion = { who = ROOT value = 25 }
			}
			NOT = {
				OR = {
					FROM = { trait = envious }
					FROM = { trait = deceitful }
					FROM = { trait = ambitious }
				}
			}
		}
		modifier = {
			factor = 1.25
			NOT = { FROM = { opinion = { who = ROOT value = 0 } } }
		}
		modifier = {
			factor = 1.5
			NOT = { FROM = { opinion = { who = ROOT value = -50 } } }

		}

			FROM = { NOT = { intrigue = 5 } }
