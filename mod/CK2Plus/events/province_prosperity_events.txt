			factor = 1.1

			factor = 1.1
			owner = { realm_learning = 40 }
		}
		modifier = {
			factor = 0.8
			has_province_modifier = province_prosperity_2
		}		
		modifier = {
			factor = 0.9

		modifier = {
			factor = 1.1
			has_province_modifier = province_poverty_1
		}
		modifier = {
			factor = 1.1

		modifier = {
			factor = 0.9

			factor = 0.9

		name = "OK"
