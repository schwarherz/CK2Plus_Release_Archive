			factor = 0.5
			has_character_flag = hunt_good_hounds
		}
		modifier = {
			factor = 1.25
			trait = craven
		}
		
	}

	option = {
		name = "EVTOPTA36000"
		trigger = {
			NOT = {
				trait = craven
			}
		}
		prestige = 5
	}
	
	option = {
		name = "EVTOPTB36000"
		trigger = {
			trait = craven
		}
		prestige = -10
	}
	option = {
		name = "EVTOPTC36000"
		trigger = {
			NOT = { trait = greedy }
			NOT = { trait = arbitrary }
			NOT = { trait = envious }
			NOT = { trait = craven }
			NOT = { trait = cruel }
		}
		add_trait = kind
	}
	
	option = {
		name = "EVTOPTD36000"
		trigger = {
			NOT = { trait = craven }
			NOT = { trait = humble }
			NOT = { trait = content }
		}
