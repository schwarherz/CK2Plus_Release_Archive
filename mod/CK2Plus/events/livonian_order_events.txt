# Hochmeister writes a letter
character_event = {
	id = 18700
	desc = "EVTDESC18700"
	picture = GFX_evt_council
	
	trigger = {
		# Teutonic Order is active
		is_title_active = d_livonian_order
		primary_title = {
			title = d_livonian_order
		}
		independent = yes
		NOT = { demesne_size = 1 }

		NOT = { has_global_flag = livonian_order_invited_to_kingdom }
		NOT = { has_global_flag = land_wanted_by_livonian_order }
		OR = {
			NOT = { has_character_flag = land_given_to_livonian_order }
			had_character_flag = { flag = land_given_to_livonian_order days = 365 }
			had_character_flag = { flag = no_land_to_livonian_order_at_the_moment = 720 }
		}

		# There is a valid province
		any_independent_ruler = {
			religion = catholic
			primary_title = {
				tier = KING
			}
			NOT = { has_character_flag = no_land_to_religious_orders }
			any_demesne_province = {
				kingdom = { title = k_lithuania }
				OR = {
					religion_group = pagan_group
					AND = {
						NOT = { religion = catholic }
						any_neighbor_province = {
							owner = {
								religion_group = pagan_group
							}
						}
					}
				}
				kingdom = {
					OR = {
						has_holder = no
						holder_scope = {
							NOT = { religion = catholic }
						}
					}
				}
			}
		}
	}
	
	mean_time_to_happen = {
		months = 12
	}
	
	immediate = {
		hidden_tooltip = {
			set_global_flag = land_wanted_by_livonian_order
		}
	}
	
	option = {
		name = "EVTOPTA18700"
		random_independent_ruler = {
			limit = {
				religion = catholic
				primary_title = {
					tier = KING
				}
				any_demesne_province = {
					kingdom = { title = k_lithuania }
					OR = {
						religion_group = pagan_group
						AND = {
							NOT = { religion = catholic }
							any_neighbor_province = {
								owner = {
									religion_group = pagan_group
								}
							}
						}
					}
					kingdom = {
						OR = {
							has_holder = no
							holder_scope = {
								NOT = { religion = catholic }
							}
						}
					}
				}
			}
			random_demesne_province = {
				limit = {
					kingdom = { title = k_lithuania }
					OR = {
						religion_group = pagan_group
						AND = {
							NOT = { religion = catholic }
							any_neighbor_province = {
								owner = {
									religion_group = pagan_group
								}
							}
						}
					}
					kingdom = {
						OR = {
							has_holder = no
							holder_scope = {
								NOT = { religion = catholic }
							}
						}
					}
				}
				set_province_flag = land_wanted_by_livonian_order
				owner = {
					letter_event = { id = 18701 days = 7 }
				}
			}
		}
	}
}

### HRE vassal wants to give land to the Teutonic Order ###

# HRE vassal wants to donate land to the Teutonic Order
province_event = {
	id = 18720
	desc = "EVTDESC18720"
	picture = GFX_evt_council
	
	only_rulers = yes
	
	trigger = {
		owner = {
			NOT = { has_character_flag = no_land_to_livonian_order }
			NOT = { has_character_flag = wants_to_give_land_to_livonian_order }
			
			war = no
			
			# The Emperor is Top liege
			top_liege = {
				primary_title = {
					title = e_hre
				}
			}

			# Teutonic Order is active
			is_title_active = d_livonian_order
			any_independent_ruler = {
				primary_title = {
					title = d_livonian_order
				}
				NOT = { demesne_size = 1 }
			}
		}
		# There is a valid province
		kingdom = {
			OR = {
				title = k_lithuania
				title = k_finland
			}
		}
	}
	
	mean_time_to_happen = {
		months = 120
		
		modifier = {
			factor = 0.75
			religion_group = pagan_group
		}
		modifier = {
			factor = 0.75
			any_neighbor_province = {
				owner = {
					religion_group = pagan_group
				}
			}
		}
	}
	
	immediate = {
		hidden_tooltip = {
			owner = { set_character_flag = wants_to_give_land_to_livonian_order }
			set_province_flag = province_to_the_livonian_order
		}
	}
	
	option = {
		name = "EVTOPTA18720"
		tooltip = {
			owner = {
				any_independent_ruler = {
					limit = {
						primary_title = {
							title = d_livonian_order
						}
					}
					ROOT = {
						county = {
							grant_title = PREVPREV
						}
					}
				}
			}
		}
		hidden_tooltip = {
			owner = {
				liege = { letter_event = { id = 18721 days = 3 } }
			}
		}
	}
	option = {
		name = "EVTOPTB8720"
		owner = { set_character_flag = no_land_to_livonian_order }
	}
}
