### Ask for a County in order to help against Pagans ###

# Hochmeister writes a letter
character_event = {
	id = 18700
	desc = "EVTDESC18700"
	picture = GFX_evt_council
	
	trigger = {
		# Teutonic Order is active
		is_title_active = d_livonian_order
		primary_title = {
			title = d_livonian_order
		}
		independent = yes
		NOT = { 
			any_demesne_title = {
				OR = {
					tier = BARON
					tier = COUNT
				}
			}
		}

		NOT = { has_global_flag = livonian_order_invited_to_kingdom }
		NOT = { has_global_flag = land_wanted_by_livonian_order }
		OR = {
			NOT = { has_character_flag = land_given_to_livonian_order }
			had_character_flag = { flag = land_given_to_livonian_order days = 365 }
			had_character_flag = { flag = no_land_to_livonian_order_at_the_moment days = 720 }
		}

		# There is a valid province
		any_independent_ruler = {
			religion = catholic
			primary_title = {
				higher_tier_than = DUKE
			}
			NOT = { has_character_flag = no_land_to_religious_orders }
			any_demesne_province = {
				kingdom = { title = k_lithuania }
				OR = {
					religion_group = pagan_group
					AND = {
						NOT = { religion = catholic }
						any_neighbor_province = {
							owner = {
								religion_group = pagan_group
							}
						}
					}
				}
				kingdom = {
					OR = {
						has_holder = no
						holder_scope = {
							NOT = { religion = catholic }
						}
					}
				}
			}
		}
	}
	
	mean_time_to_happen = {
		months = 12
	}
	
	immediate = {
		hidden_tooltip = {
			set_global_flag = land_wanted_by_livonian_order
		}
	}
	
	option = {
		name = "EVTOPTA18700"
		random_independent_ruler = {
			limit = {
				religion = catholic
				primary_title = {
					higher_tier_than = DUKE
				}
				any_demesne_province = {
					kingdom = { title = k_lithuania }
					OR = {
						religion_group = pagan_group
						AND = {
							NOT = { religion = catholic }
							any_neighbor_province = {
								owner = {
									religion_group = pagan_group
								}
							}
						}
					}
					kingdom = {
						OR = {
							has_holder = no
							holder_scope = {
								NOT = { religion = catholic }
							}
						}
					}
				}
			}
			random_demesne_province = {
				limit = {
					kingdom = { title = k_lithuania }
					OR = {
						religion_group = pagan_group
						AND = {
							NOT = { religion = catholic }
							any_neighbor_province = {
								owner = {
									religion_group = pagan_group
								}
							}
						}
					}
					kingdom = {
						OR = {
							has_holder = no
							holder_scope = {
								NOT = { religion = catholic }
							}
						}
					}
				}
				set_province_flag = province_wanted_by_livonian_order
				owner = {
					letter_event = { id = 18701 days = 7 }
				}
			}
		}
	}
}

# Ask permission from the King
letter_event = {
	id = 18701
	desc = "EVTDESC18701"
	
	is_triggered_only = yes
	
	option = {
		name = "EVTOPTA18501"
		ai_chance = { factor = 100 }
		prestige = 50
		piety = 50
		random_demesne_province = {
			limit = { has_province_flag = province_wanted_by_livonian_order }
			county = {
				grant_title = FROM
				FROM = {
					add_character_modifier = { name = "holy_truce_timer" duration = 3650 }
					gain_settlements_under_title = {
						title = PREV
						enemy = ROOT
					}
				}
				hidden_tooltip = { add_building = ca_order_house }
			}
		}
		FROM = {
			opinion = {
				modifier = opinion_granted_county_to_holy_order
				who = ROOT
				years = 100
			}
		}
		religion_head = {
			opinion = {
				modifier = opinion_exalted
				who = ROOT
				years = 10
			}
		}
		hidden_tooltip = {
			FROM = {
				random_demesne_province = {
					limit = { has_province_flag = province_wanted_by_livonian_order }
					province_event = { id = 18702 days = 1 }
				}
			}
		}
		set_character_flag = land_given_to_livonian_order
	}
	option = {
		name = "EVTOPTB18501"
		ai_chance = { factor = 0 }
		set_character_flag = no_land_to_livonian_order_at_the_moment
	}
	option = {
		name = "EVTOPTC18501"
		ai_chance = { factor = 0 }
		piety = -25
		FROM = {
			opinion = {
				modifier = opinion_angry
				who = ROOT
				years = 20
			}
		}
		religion_head = {
			opinion = {
				modifier = opinion_disappointed
				who = ROOT
				years = 10
			}
		}
		hidden_tooltip = {
			FROM = { letter_event = { id = 18800 days = 1 } }
		}
		set_character_flag = no_land_to_religious_orders
		hidden_tooltip = {
			FROM = {
				random_demesne_province = {
					limit = { has_province_flag = province_wanted_by_livonian_order }
					clr_province_flag = province_wanted_by_livonian_order
				}
			}
		}		
		clr_global_flag = land_wanted_by_livonian_order
	}
}
