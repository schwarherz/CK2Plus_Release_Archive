#Syphilitic
character_event = {
	id = 6000
	desc = "EVTDESC6000"
	picture = "GFX_evt_illness"
	
	capable_only = yes
	
	trigger = {
		age = 16
		NOT = {
			trait = syphilitic
		}
		OR = {
			NOT = { trait = chaste }
			spouse = { 
				is_alive = yes
				is_abroad = no
				trait = syphilitic 
			}
		}
		NOT = {
			trait = celibate
		}
		NOT = {
			trait = infirm
		}
	}
	
	mean_time_to_happen = {
		months = 10000
		
		modifier = {
			factor = 0.01
			spouse = { 
				trait = syphilitic 
			}
		}
		modifier = {
			factor = 0.01
			lover = { 
				trait = syphilitic 
			}
		}		
		modifier = {
			factor = 0.5
			in_command = yes
		}		
		modifier = {
			factor = 0.5
			trait = lustful
		}
		modifier = {
			factor = 0.6
			trait = hedonist
		}
		
		modifier = {
			factor = 5.0
			trait = chaste
		}
	}
	
	option = {
		name = "EVTOPTA6000"
		add_trait = syphilitic
	}
}

		months = 120

		modifier = {

			trait = ill
		}
		modifier = {

			trait = pneumonic
		}		
		modifier = {
			factor = 0.2
			has_character_modifier = the_oubliette
		}
		modifier = {
			factor = 0.4
			has_character_modifier = the_dungeon
		}
		modifier = {
			factor = 0.75
			NOT = { base_health = 2 }
		}		
		modifier = {
			factor = 0.75
			NOT = { base_health = 4 }
		}		
		modifier = {
			factor = 1.5
			base_health = 6
		}
		modifier = {
			factor = 2.0
			base_health = 7
		}
		modifier = {
			factor = 2.0
			base_health = 8
		}
	}
	
	option = {
		name = "EVTOPTA6600"
		random_list = {
			20 = { death = { death_reason = death_accident } }
			80 = {
				remove_trait = wounded
				add_trait = maimed			
			}
		}	
		set_character_flag = limb_removed
	}
